Usage

nvm

append to .bash_profile, and `source .bash_profile`, also you can using .bashrc file as well 

  NVM_NODEJS_ORG_MIRROR=<%= baseUrl %>/node

or you can just run the cmd in cli:

  NVM_NODEJS_ORG_MIRROR=<%= baseUrl %>/node nvm install 4

if you'r using the latest version of nvm, be careful of the IOJS mirror setting
if you set up `NVM_IOJS_ORG_MIRROR`, the downloading action will always target to iojs-pkg
so do not set up this env value only if you need iojs 

  NVM_IOJS_ORG_MIRROR=<%= baseUrl %>/iojs nvm install 2

nvmw

  set "NVMW_NODEJS_ORG_MIRROR=<%= baseUrl %>/node"
  set "NVMW_IOJS_ORG_MIRROR=<%= baseUrl %>/iojs"
  set "NVMW_NPM_MIRROR=<%= baseUrl %>/npm"

  nvmw install node-v0.12.2
  nvmw install iojs-v1.6.3

npm

  npm install phantomjs --phantomjs_cdnurl=<%= baseUrl %>/phantomjs
  npm install chromedriver --chromedriver_cdnurl=<%= baseUrl %>/chromedriver
  npm install operadriver --operadriver_cdnurl=<%= baseUrl %>/operadriver

electron

  ELECTRON_MIRROR=<%= baseUrl %>/electron/ npm install -g electron

electron-builder

  ELECTRON_BUILDER_BINARIES_MIRROR=<%= baseUrl %>/electron-builder-binaries/ electron-builder build

electron-rebuild

  electron-rebuild -d=<%= baseUrl %>/atom-shell
  electron-rebuild -d=<%= baseUrl %>/atom-shell -f -w $module_with_c++_code

selenium-standalone

  # choose selenium version
  selenium-standalone install --version=2.43.0 --baseURL=<%= baseUrl %>/selenium

  # choose chrome driver version
  selenium-standalone install --drivers.chrome.version=2.12 --drivers.chrome.baseURL=<%= baseUrl %>/chromedriver

  # choose ie driver architecture
  selenium-standalone start --drivers.ie.arch=ia32 --drivers.ie.baseURL=<%= baseUrl %>/selenium

node-sass

  SASS_BINARY_SITE=<%= baseUrl %>/node-sass npm install node-sass

node-sqlite3

  npm install sqlite3 --node_sqlite3_binary_host_mirror=<%= baseUrl %>

windows-build-tools

  set "PYTHON_MIRROR=<%= baseUrl %>/python" && npm install --global --production windows-build-tools

node-inspector

  # npm version
  npm install node-inspector --profiler_binary_host_mirror=<%= baseUrl %>/node-inspector/
  # yarn version
  npm_config_profiler_binary_host_mirror=<%= baseUrl %>/node-inspector/ yarn add node-inspector
  # yarn global version
  npm_config_profiler_binary_host_mirror=<%= baseUrl %>/node-inspector/ yarn global add node-inspector
